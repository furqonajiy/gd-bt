# =========================================================================
# ADAPTIVE REGIME AUTO-SWITCH  (auto --adaptive)            | tag ADPT
# ONE live executor that classifies the current volatility regime each cycle
# (R1quiet/R2bull/R3strong/R4parab, from smoothed M15 ATR + trend) and runs that
# regime's PUBLISHED champion config (CHAMPION_<regime>.json under
# --champions-dir). When a regime has no champion yet it FALLS BACK to the SC24
# strategy flags below (the live incumbent), so you are never without a config.
#
# Feed: scalper24 (the dense intraday feed SC24 is the R4 champion of). --adaptive
#   swaps the STRATEGY CONFIG by regime; it does NOT switch the signal feed, so
#   run it on this one feed. (Per-feed switching is a future enhancement.)
# Champions source: ./champions/CHAMPION_<regime>.json (committed on main). The
#   three completed regimes all promote the SAME config: R2bull = R3strong =
#   R4parab = SC24T24E8 (SC24 + entry_count 6->8, tp1-lock-delay stays 24) -- it is
#   #1 on the reliable forward-fit metrics (OOS AND fixed-lot edge) in every one of
#   them (R2 OOS 52,831 / R3 110,907 / R4 15,032) at DD<=40%; the universal lever is
#   more entries (e8>e7>e6 on OOS). It supersedes the earlier R4 pick SC24T15E6,
#   which led only on the compounded net+bonus mirage. R1quiet is still an SC24
#   placeholder until the sweep advances to it. As each regime gets a validated
#   winner, drop its CHAMPION_<regime>.json into champions/ and both the backtest
#   (--adaptive) and this live executor pick it up.
# compounded $ are model upper bounds — they rank configs, not forecast money.
# =========================================================================

cd C:\Git\github.com\furqonajiy\xauusd-backtest
conda activate trading

# ===== 1. SIGNAL AUTO GENERATOR =====
# LIVE SIGNAL GENERATOR (single command; logs each new signal). One process, no
# shell loop: refetches the current month + regenerates ONLY when a new closed
# M1 bar exists, narrows to the 2 most recent months + a rolling 3-day window.
python tools/live_feed_loop.py `
  --family scalper `
  --interval 30 `
  --gen-start-days 3 `
  --gen-recent-months 2 `
  --mt5-symbol XAUUSD `
  --mt5-server-offset 3 `
  -- --charts data/XAUUSD_M1_*_ELEV8.csv --output generated/self_scalper24_live.txt --session-start 0 --session-end 0 --signal-tz 7

# ===== 2. BACKTEST CLI =====
# WHERE PER-REGIME PARAMS LIVE: champions/CHAMPION_<regime>.json -- ONE file per
# regime (R1quiet / R2bull / R3strong / R4parab); --adaptive reads them. TODAY
# R2bull = R3strong = R4parab = SC24T24E8 (entry 8) -- the OOS/edge winner promoted
# for all three trending regimes; only R1quiet is still an SC24 placeholder, so it
# backtests as SC24 until the sweep advances to it (or you hand-edit its CHAMPION file).
# To give a regime DIFFERENT parameters, edit the "config" block in
# champions/CHAMPION_<regime>.json -- do NOT duplicate params across commands.
# Regime windows (the sweep's slices): R1quiet 2021-11..2023-09 | R2bull
#   2023-10..2024-12 | R3strong 2025 | R4parab 2026.
# STEP 2a — (re)generate the FULL archive once (all history):
python tools/generate_scalper_signals.py --charts data/XAUUSD_M1_*_ELEV8.csv --output generated/self_scalper24.txt --start 2021-11-01 --session-start 0 --session-end 0 --signal-tz 7
# STEP 2b — ADAPTIVE backtest. Run FULL history (every regime switches in turn --
# the Summary's Regime column shows R1->R4 across the months), OR bound the dates
# to backtest ONE regime in isolation (only the last 3 flags change -- see below):
python tools/backtest_explicit.py `
  --signals generated/self_scalper24.txt `
  --charts data/XAUUSD_M1_*_ELEV8.csv `
  --max-drawdown-limit-pct 50 `
  --progress-interval-seconds 30 `
  --sync-charts true `
  --sizing-mode risk `
  --lot 0.01 `
  --risk 0.01 `
  --minimum-lot 0.01 `
  --lot-step 0.01 `
  --bonus-per-closed-lot 3.0 `
  --entries 6 `
  --entry-ladder range_to_sl `
  --entry-sl-gap 0.5 `
  --activation-delay 2 `
  --pending-expiry 180 `
  --max-hold 240 `
  --sl-multiplier 2.1 `
  --final-target TP3 `
  --lock-after-tp1 true `
  --lock-after-tp2 true `
  --tp1-lock-delay-minutes 24 `
  --tp2-lock-delay-minutes 2 `
  --profit-lock-mode tp_levels `
  --bep-trigger-distance 3.0 `
  --tp1-lock-fraction 0.5 `
  --tp2-lock-target TP1 `
  --runner-after-tp3 false `
  --tp3-lock-target TP2 `
  --trailing-open-distance 0.0 `
  --trailing-close-distance 0.0 `
  --trailing-close-after-stage 0 `
  --adaptive true `
  --champions-dir champions `
  --adaptive-window-days 20 `
  --output-dir reports/ADAPTIVE_ALL `
  --initial-capital 5000 `
  --start-date 2021-11-01

# Per-regime isolation -- rerun the SAME command above, changing ONLY these 3 flags:
#   R1quiet  : --start-date 2021-11-01 --end-date 2023-09-30 --output-dir reports/REGIME_R1quiet
#   R2bull   : --start-date 2023-10-01 --end-date 2024-12-31 --output-dir reports/REGIME_R2bull
#   R3strong : --start-date 2025-01-01 --end-date 2025-12-31 --output-dir reports/REGIME_R3strong
#   R4parab  : --start-date 2026-01-01                       --output-dir reports/REGIME_R4parab

# STEP 2c — EXPLICIT params for a custom range (NO --adaptive): to test a SPECIFIC
# config on a SPECIFIC period directly, drop --adaptive/--champions-dir and set the
# strategy flags yourself + --start-date/--end-date. e.g. "config A for 2021-2022,
# config B for 2022-2025" = two runs, each with its own flags and date window:
#   python tools/backtest_explicit.py ...config-A-flags... --start-date 2021-01-01 --end-date 2022-12-31 --output-dir reports/A_2021_2022 --initial-capital 5000
#   python tools/backtest_explicit.py ...config-B-flags... --start-date 2022-01-01 --end-date 2025-12-31 --output-dir reports/B_2022_2025 --initial-capital 5000
