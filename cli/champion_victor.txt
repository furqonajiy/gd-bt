# ===== 1. TELEGRAM LISTENER =====
# Ingests Victor's Telegram channel into victor_signals.txt -- this IS this CLI's
# live signal source.
#    First-time setup only: `python listeners\telegram\listener.py list-chats`
#    to find the channel id, then paste it into listener_config.json.
python listeners\telegram\listener.py --signals-file victor_signals.txt

# ===== 2. LIVE LOOP SIGNAL GENERATOR =====
# Victor's "live loop" is the provider filter in --watch mode: it regenerates
# signals/victor_live.txt from victor_signals.txt on every change.
#    `--preset all` keeps EVERY Victor signal (no hour/side filtering), so the
#    live feed is exactly Victor's signals and matches the backtest below.
python tools/live_provider_signal_filter.py `
  --input victor_signals.txt `
  --output signals/victor_live.txt `
  --preset all `
  --watch `
  --interval 2

# ===== 3. LIVE AUTO EXECUTOR =====
# Reads the filtered feed (section 2), places + manages real orders on MT5.
# Strategy parameters are byte-identical to the parity backtests (sections 5-9); only the backtest-only locked-exit slippage varies by era there, never sent live.
# Flags omitted because they do NOT affect live execution (auto_explicit defaults
# them): --initial-capital (live sizes off REAL MT5 equity), --bonus-per-closed-lot
# (backtest scoring only), --tp3-lock-target (only TP2). Also dropped the flag
# already at its default: --trailing-close-after-stage 0. KEPT because they are
# NON-default and Victor needs them: --replace-missing-entries / --reopen-missing-
# positions / --apply-signal-edits / --signal-overrides-file.
python tools/auto_explicit.py `
  --signals signals/victor_live.txt `
  --positions-json positions_victor.json `
  --strategy-tag VIC `
  --forensic-log forensic_victor.jsonl `
  --notifications notifications_victor.jsonl `
  --watch-interval 3 `
  --mt5-symbol XAUUSD `
  --mt5-server-offset 3 `
  --mt5-history-bars 5000 `
  --sizing-mode risk `
  --lot 0.01 `
  --risk 0.025 `
  --minimum-lot 0.01 `
  --lot-step 0.01 `
  --entries 8 `
  --entry-ladder range_to_sl `
  --entry-sl-gap 0.5 `
  --activation-delay 2 `
  --pending-expiry 180 `
  --max-hold 240 `
  --sl-multiplier 2.1 `
  --final-target TP3 `
  --lock-after-tp1 true `
  --lock-after-tp2 true `
  --tp1-lock-delay-minutes 24 `
  --tp2-lock-delay-minutes 2 `
  --profit-lock-mode tp_levels `
  --bep-trigger-distance 3 `
  --tp1-lock-fraction 0.5 `
  --tp2-lock-target TP1 `
  --runner-after-tp3 false `
  --trailing-open-distance 0 `
  --trailing-close-distance 0 `
  --replace-missing-entries true `
  --reopen-missing-positions true `
  --apply-signal-edits true `
  --signal-overrides-file signal_overrides.jsonl

# ===== 4. SIGNAL GENERATOR (before backtest) =====

  --sl-multiplier 2.1 `
  --final-target TP3 `
  --lock-after-tp1 true `
  --lock-after-tp2 true `
  --tp1-lock-delay-minutes 24 `
  --tp2-lock-delay-minutes 2 `
  --profit-lock-mode tp_levels `
  --bep-trigger-distance 3 `
  --tp1-lock-fraction 0.5 `
  --tp2-lock-target TP1 `
