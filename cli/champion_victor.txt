# ===== 1. TELEGRAM LISTENER =====
# Ingests Victor's Telegram channel into victor_signals.txt -- this IS this CLI's
# live signal source.
#    First-time setup only: `python listeners\telegram\listener.py list-chats`
#    to find the channel id, then paste it into listener_config.json.
python listeners\telegram\listener.py --signals-file victor_signals.txt

# ===== 2. LIVE LOOP SIGNAL GENERATOR =====
# Victor's "live loop" is the provider filter in --watch mode: it regenerates
# signals/victor_live.txt from victor_signals.txt on every change.
#    `--preset all` keeps EVERY Victor signal (no hour/side filtering), so the
#    live feed is exactly Victor's signals and matches the backtest below.
python tools/live_provider_signal_filter.py `
  --input victor_signals.txt `
  --output signals/victor_live.txt `
  --preset all `
  --watch `
  --interval 2

# ===== 3. SIGNAL GENERATOR (before backtest) =====
# N/A -- there is no generation step for Victor. The backtest reads
# victor_signals.txt directly (the listener output IS the signal source); sections
# 4/5 below pass it as --signals.

# ===== 4. BACKTEST FROM 2026-01 (current regime) =====
# --lock-tp1-exit-slippage 2.0 / --lock-tp2-exit-slippage 1.0 model the LOCKED-exit
#   give-back so the backtest matches live. Backtest-only; never live. Strategy
#   params are byte-identical to the auto runner (section 6).
python tools/backtest_explicit.py `
  --signals victor_signals.txt `
  --charts data/XAUUSD_M1_*_ELEV8.csv `
  --max-drawdown-limit-pct 50 `
  --progress-interval-seconds 30 `
  --sizing-mode risk `
  --lot 0.01 `
  --risk 0.025 `
  --minimum-lot 0.01 `
  --lot-step 0.01 `
  --bonus-per-closed-lot 3 `
  --entries 8 `
  --entry-ladder range_to_sl `
  --entry-sl-gap 0.5 `
  --activation-delay 2 `
  --pending-expiry 180 `
  --max-hold 240 `
  --sl-multiplier 2.1 `
  --final-target TP3 `
  --lock-after-tp1 true `
  --lock-after-tp2 true `
  --tp1-lock-delay-minutes 24 `
  --tp2-lock-delay-minutes 2 `
  --profit-lock-mode tp_levels `
  --bep-trigger-distance 3 `
  --tp1-lock-fraction 0.5 `
  --tp2-lock-target TP1 `
  --runner-after-tp3 false `
  --tp3-lock-target TP2 `
  --trailing-open-distance 0 `
  --trailing-close-distance 0 `
  --lock-tp1-exit-slippage 2.0 `
  --lock-tp2-exit-slippage 1.0 `
  --trailing-close-after-stage 0 `
  --output-dir reports/VIC_202601 `
  --initial-capital 50000 `
  --start-date 2026-01-01
