# ===== 1. TELEGRAM LISTENER =====
# Ingests Victor's Telegram channel into victor_signals.txt. This IS this CLI's
# live signal source.
#    First-time setup only: `python listener\telegram_listener.py list-chats`
#    to find the channel id, then paste it into listener_config.json.
python listener\telegram_listener.py --signals-file victor_signals.txt

# ===== 2. SIGNAL AUTO GENERATOR =====
# Victor signal filter -> reads victor_signals.txt (the listener's raw feed) and
# writes generated/victor_live.txt, the Victor feed the live auto trades.
#    `--preset all` keeps EVERY Victor signal (no hour/side filtering), so the
#    live feed is exactly Victor's signals and matches the backtest below.
python tools/live_provider_signal_filter.py `
  --input victor_signals.txt `
  --output generated/victor_live.txt `
  --preset all `
  --watch `
  --interval 2

# ===== 3. BACKTEST CLI =====
# Parity backtest (reference; same strategy parameters as the auto runner).
python tools/backtest_explicit.py `

  --charts data/XAUUSD_M1_*_ELEV8.csv `

  --pending-expiry 180 `
  --max-hold 240 `
  --sl-multiplier 2.1 `
  --final-target TP3 `
  --lock-after-tp1 true `
  --lock-after-tp2 true `
  --tp1-lock-delay-minutes 24 `
  --tp2-lock-delay-minutes 2 `
  --profit-lock-mode tp_levels `
  --bep-trigger-distance 3 `
  --tp1-lock-fraction 0.5 `
